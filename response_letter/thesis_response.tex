\usepackage[a4paper,left=37mm,right=30mm,top=35mm,bottom=30mm]{geometry}

% add the cross reference links
\usepackage{hyperref}
\hypersetup{
	colorlinks = false,
	linkbordercolor={0 0 0},
}

% for citation management
\usepackage[style=authoryear, sorting=nty, backend=bibtex]{biblatex}
\addbibresource{response.bib}
\usepackage{url}


% other packages
\usepackage{xcolor}  % highlight some text

% package needed for optional arguments
\usepackage{xifthen}
% define counters for examiners and their points
\newcounter{examiner}
\setcounter{examiner}{0}
\newcounter{point}[examiner]
\setcounter{point}{0}

% This refines the format of how the examiner/point reference will appear.
\renewcommand{\thepoint}{P\,\theexaminer.\arabic{point}} 

% command declarations for examiner points and our responses
\newcommand{\examinersection}{\stepcounter{examiner} \bigskip \hrule
                  \section{Examiner \theexaminer}}

\newenvironment{point}
   {\refstepcounter{point} \bigskip \noindent {\textbf{Examiner~Point~\thepoint} } ---\ }
   {\par }

\newcommand{\shortpoint}[1]{\refstepcounter{point}  \bigskip \noindent 
	{\textbf{Examiner~Point~\thepoint} } ---~#1\par }

\newenvironment{reply}
   {\medskip \noindent \begin{sf}\textbf{Reply}:\  }
   {\medskip \end{sf}}

\newcommand{\shortreply}[2][]{\medskip \noindent \begin{sf}\textbf{Reply}:\  #2
	\ifthenelse{\equal{#1}{}}{}{ \hfill \footnotesize (#1)}%
	\medskip \end{sf}}

\begin{document}
	
\title{\large \textbf{Response Letter for Article: \\ Name of the Article}}
\author{Your Name \\ \href{mailto:your email}{your email}}

\maketitle

% Start the cover letter of the response letter

\noindent Dear Editor and Reviewers,

\bigskip

% General intro text goes here
\noindent I am very grateful for the critical assessment and insightful comments of my article "\textbf{The name of the article}" from reviewers. I have revised the article and addressed each of the questions raised by reviewers in this response letter. The major points are summarized below:

% Summarize the major edits
\begin{enumerate}
	\item Point 1
	\item Point 2
	\item Point 3
\end{enumerate}

\noindent Please find below the responses to the specific comments raised by each reviewer.

\bigskip

\noindent Yours sincerely,

\noindent Your Name

\noindent \href{mailto:your email}{your email}

\newpage

% Let's start point-by-point with examiner 1
\examinersection

\subsection{Comments from Reviewer One}

\begin{point}

\end{point}

% Our reply
\begin{reply}

\end{reply}

\printbibliography[keyword={examiner1},title={Reference}]

\pagebreak

% Begin a new examiner section
\examinersection

\subsection{Comments from Reviewer Two}

\end{point}

\begin{reply}
\lipsum[4] \parencite{Chainey2008utility}
\end{reply}

\printbibliography[keyword={examiner2}, title={Reference}]

\pagebreak

% Begin a new examiner section
\examinersection

\subsection{General Perception}

\lipsum[5]

\subsection{Comments from Reviewer Three}

\begin{reply}

\end{reply}

\end{document}
